``` python 
    ans=[]

     left=0
```

``` python 
def expand_string(s):
    expanded_str = ""
    i = 0
    
    while i < len(s):
        # Extract the character
        char = s[i]
        i += 1
        
        # Extract the number
        num_str = ""
        while i < len(s) and s[i].isdigit():
            num_str += s[i]
            i += 1
            
        # Convert the number to an integer
        num = int(num_str)
        
        # Append the character 'num' times to the result
        expanded_str += char * num
    
    return expanded_str
```
